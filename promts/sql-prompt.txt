-- ============================================
-- ELECTRONICS INVENTORY MANAGEMENT SYSTEM
-- Enhanced Schema with Offline & Sync Support
-- ============================================

-- ========== CORE TABLES ==========

-- ADDRESS TABLE
CREATE TABLE `address` (
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `postal_code` varchar(20) NOT NULL,
  `address_id` varchar(36) PRIMARY KEY,
  `created_by` varchar(36) DEFAULT NULL,
  `updated_by` varchar(36) DEFAULT NULL,
  `city` varchar(100) NOT NULL,
  `country` varchar(100) NOT NULL,
  `state` varchar(100) NOT NULL,
  `address_line1` varchar(255) NOT NULL,
  `address_line2` varchar(255) DEFAULT NULL,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'SYNCED'
);

-- CATEGORY TABLE
CREATE TABLE `category` (
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `category_id` varchar(36) PRIMARY KEY,
  `created_by` varchar(36) DEFAULT NULL,
  `updated_by` varchar(36) DEFAULT NULL,
  `name` varchar(255) NOT NULL,
  `description` text DEFAULT NULL,
  `category_type` ENUM('PRODUCT', 'SPARE_PART', 'BOTH') DEFAULT 'PRODUCT',
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'SYNCED'
);

-- LOCATION TABLE
CREATE TABLE `location` (
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `created_by` varchar(36) DEFAULT NULL,
  `location_id` varchar(36) PRIMARY KEY,
  `updated_by` varchar(36) DEFAULT NULL,
  `name` varchar(100) NOT NULL,
  `location_type` ENUM('STORE', 'WORKSHOP', 'WAREHOUSE') DEFAULT 'STORE',
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'SYNCED'
);

-- CUSTOMER TABLE
CREATE TABLE `customer` (
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `phone` varchar(15) DEFAULT NULL,
  `customer_id` varchar(36) PRIMARY KEY,
  `address_id` varchar(36) DEFAULT NULL,
  `created_by` varchar(36) DEFAULT NULL,
  `updated_by` varchar(36) DEFAULT NULL,
  `email` varchar(100) DEFAULT NULL,
  `name` varchar(100) NOT NULL,
  `customer_type` ENUM('RETAIL', 'MAINTENANCE', 'BOTH') DEFAULT 'RETAIL',
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'SYNCED',
  FOREIGN KEY (`address_id`) REFERENCES `address`(`address_id`)
);

-- ITEM TABLE (Enhanced)
CREATE TABLE `item` (
  `item_id` varchar(36) PRIMARY KEY,
  `name` varchar(100) NOT NULL,
  `description` text DEFAULT NULL,
  `price` decimal(10,2) NOT NULL,
  `cost` decimal(10,2) DEFAULT 0.00,
  `category_id` varchar(36) DEFAULT NULL,
  `location_id` varchar(36) DEFAULT NULL,
  `item_type` ENUM('PRODUCT', 'SPARE_PART') DEFAULT 'PRODUCT',
  `is_for_maintenance` boolean DEFAULT FALSE,
  `min_stock_level` int DEFAULT 5,
  `reorder_quantity` int DEFAULT 10,
  `stock_location` ENUM('STORE', 'WORKSHOP', 'BOTH') DEFAULT 'BOTH',
  `sku` varchar(50) UNIQUE,
  `barcode` varchar(100) DEFAULT NULL,
  `weight` decimal(8,2) DEFAULT NULL,
  `dimensions` varchar(50) DEFAULT NULL,
  `manufacturer` varchar(100) DEFAULT NULL,
  `warranty_period` int DEFAULT NULL, -- in months
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `created_by` varchar(36) DEFAULT NULL,
  `updated_by` varchar(36) DEFAULT NULL,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'SYNCED',
  FOREIGN KEY (`category_id`) REFERENCES `category`(`category_id`),
  FOREIGN KEY (`location_id`) REFERENCES `location`(`location_id`)
);

-- CUSTOMER_DEVICE TABLE (Customer-owned items for repairs)
CREATE TABLE `customer_device` (
  `device_id` varchar(36) PRIMARY KEY,
  `customer_id` varchar(36) NOT NULL,
  `catalog_item_id` varchar(36) DEFAULT NULL, -- Optional reference to catalog model
  `brand` varchar(100) DEFAULT NULL,
  `model` varchar(100) DEFAULT NULL,
  `serial_number` varchar(100) DEFAULT NULL,
  `purchase_date` date DEFAULT NULL,
  `warranty_expiry` date DEFAULT NULL,
  `notes` text DEFAULT NULL,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'SYNCED',
  FOREIGN KEY (`customer_id`) REFERENCES `customer`(`customer_id`),
  FOREIGN KEY (`catalog_item_id`) REFERENCES `item`(`item_id`)
);

-- INVENTORY TABLE (For detailed stock tracking)
CREATE TABLE `inventory` (
  `inventory_id` varchar(36) PRIMARY KEY,
  `item_id` varchar(36) NOT NULL,
  `location_id` varchar(36) NOT NULL,
  `quantity` int NOT NULL DEFAULT 0,
  `reserved_quantity` int DEFAULT 0,
  `available_quantity` int GENERATED ALWAYS AS (`quantity` - `reserved_quantity`) VIRTUAL,
  `batch_number` varchar(50) DEFAULT NULL,
  `expiry_date` date DEFAULT NULL,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `created_by` varchar(36) DEFAULT NULL,
  `updated_by` varchar(36) DEFAULT NULL,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'SYNCED',
  FOREIGN KEY (`item_id`) REFERENCES `item`(`item_id`),
  FOREIGN KEY (`location_id`) REFERENCES `location`(`location_id`),
  UNIQUE KEY `unique_item_location` (`item_id`, `location_id`)
);

-- SUPPLIER TABLE
CREATE TABLE `supplier` (
  `supplier_id` varchar(36) PRIMARY KEY,
  `name` varchar(100) NOT NULL,
  `contact_info` varchar(255) DEFAULT NULL,
  `email` varchar(100) DEFAULT NULL,
  `phone` varchar(15) DEFAULT NULL,
  `address_id` varchar(36) DEFAULT NULL,
  `supplier_type` ENUM('PRODUCT', 'SPARE_PART', 'BOTH') DEFAULT 'BOTH',
  `payment_terms` varchar(100) DEFAULT NULL,
  `lead_time_days` int DEFAULT 7,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `created_by` varchar(36) DEFAULT NULL,
  `updated_by` varchar(36) DEFAULT NULL,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'SYNCED',
  FOREIGN KEY (`address_id`) REFERENCES `address`(`address_id`)
);

-- ITEM_SUPPLIER TABLE
CREATE TABLE `item_supplier` (
  `item_supplier_id` varchar(36) PRIMARY KEY,
  `item_id` varchar(36) NOT NULL,
  `supplier_id` varchar(36) NOT NULL,
  `supplier_sku` varchar(50) NOT NULL,
  `supplier_price` decimal(10,2) DEFAULT NULL,
  `minimum_order_quantity` int DEFAULT 1,
  `is_preferred` boolean DEFAULT FALSE,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `created_by` varchar(36) DEFAULT NULL,
  `updated_by` varchar(36) DEFAULT NULL,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'SYNCED',
  FOREIGN KEY (`item_id`) REFERENCES `item`(`item_id`),
  FOREIGN KEY (`supplier_id`) REFERENCES `supplier`(`supplier_id`),
  UNIQUE KEY `unique_item_supplier` (`item_id`, `supplier_id`)
);

-- USER TABLE (Employees)
CREATE TABLE `user` (
  `user_id` varchar(36) PRIMARY KEY,
  `username` varchar(50) NOT NULL UNIQUE,
  `password` varchar(100) NOT NULL,
  `email` varchar(100) DEFAULT NULL,
  `phone` varchar(15) DEFAULT NULL,
  `first_name` varchar(50) NOT NULL,
  `last_name` varchar(50) NOT NULL,
  `address_id` varchar(36) DEFAULT NULL,
  `department` ENUM('STORE', 'WORKSHOP', 'ADMIN', 'MANAGEMENT') DEFAULT 'STORE',
  `position` varchar(50) DEFAULT NULL,
  `is_active` boolean DEFAULT TRUE,
  `last_login` datetime DEFAULT NULL,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `created_by` varchar(36) DEFAULT NULL,
  `updated_by` varchar(36) DEFAULT NULL,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'SYNCED',
  FOREIGN KEY (`address_id`) REFERENCES `address`(`address_id`)
);

-- ROLE TABLE
CREATE TABLE `role` (
  `role_id` varchar(36) PRIMARY KEY,
  `role_name` varchar(50) NOT NULL UNIQUE,
  `description` text DEFAULT NULL,
  `permissions` json DEFAULT NULL,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `created_by` varchar(36) DEFAULT NULL,
  `updated_by` varchar(36) DEFAULT NULL,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'SYNCED'
);

-- USER_ROLE TABLE
CREATE TABLE `user_role` (
  `user_role_id` varchar(36) PRIMARY KEY,
  `user_id` varchar(36) NOT NULL,
  `role_id` varchar(36) NOT NULL,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `created_by` varchar(36) DEFAULT NULL,
  `updated_by` varchar(36) DEFAULT NULL,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'SYNCED',
  FOREIGN KEY (`user_id`) REFERENCES `user`(`user_id`),
  FOREIGN KEY (`role_id`) REFERENCES `role`(`role_id`),
  UNIQUE KEY `unique_user_role` (`user_id`, `role_id`)
);

-- REPAIR_SERVICE TABLE (External repair services)
CREATE TABLE `repair_service` (
  `repair_service_id` varchar(36) PRIMARY KEY,
  `name` varchar(100) NOT NULL,
  `contact_info` varchar(255) DEFAULT NULL,
  `address_id` varchar(36) DEFAULT NULL,
  `specialization` varchar(100) DEFAULT NULL,
  `service_rating` decimal(3,2) DEFAULT NULL,
  `is_active` boolean DEFAULT TRUE,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `created_by` varchar(36) DEFAULT NULL,
  `updated_by` varchar(36) DEFAULT NULL,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'SYNCED',
  FOREIGN KEY (`address_id`) REFERENCES `address`(`address_id`)
);

-- ========== TRANSACTION TABLES ==========

-- MAINTENANCE_TICKET TABLE
CREATE TABLE `maintenance_ticket` (
  `ticket_id` varchar(36) PRIMARY KEY,
  `ticket_number` varchar(20) UNIQUE,
  `customer_id` varchar(36) NOT NULL,
  `customer_device_id` varchar(36) NOT NULL, -- Customer-owned device being repaired
  `technician_id` varchar(36) DEFAULT NULL,
  `status` ENUM('PENDING', 'DIAGNOSIS', 'WAITING_PARTS', 'IN_PROGRESS', 'TESTING', 'COMPLETED', 'DELIVERED', 'CANCELLED') DEFAULT 'PENDING',
  `problem_description` text NOT NULL,
  `diagnosis` text DEFAULT NULL,
  `estimated_cost` decimal(10,2) DEFAULT NULL,
  `estimated_completion` datetime DEFAULT NULL,
  `repair_cost` decimal(10,2) DEFAULT NULL,
  `labor_cost` decimal(10,2) DEFAULT NULL,
  `labor_hours` decimal(5,2) DEFAULT NULL,
  `total_cost` decimal(10,2) GENERATED ALWAYS AS (`repair_cost` + `labor_cost`) VIRTUAL,
  `payment_status` ENUM('PENDING', 'PAID', 'PARTIAL', 'INVOICED') DEFAULT 'PENDING',
  `priority` ENUM('LOW', 'MEDIUM', 'HIGH', 'URGENT') DEFAULT 'MEDIUM',
  `warranty_status` ENUM('IN_WARRANTY', 'OUT_OF_WARRANTY', 'EXTENDED') DEFAULT 'OUT_OF_WARRANTY',
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `completed_at` datetime DEFAULT NULL,
  `delivered_at` datetime DEFAULT NULL,
  `created_by` varchar(36) DEFAULT NULL,
  `updated_by` varchar(36) DEFAULT NULL,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'PENDING',
  `location_id` varchar(36) NOT NULL,
  FOREIGN KEY (`customer_id`) REFERENCES `customer`(`customer_id`),
  FOREIGN KEY (`customer_device_id`) REFERENCES `customer_device`(`device_id`),
  FOREIGN KEY (`technician_id`) REFERENCES `user`(`user_id`),
  FOREIGN KEY (`location_id`) REFERENCES `location`(`location_id`)
);

-- SALE TABLE (Header for retail & maintenance invoices)
CREATE TABLE `sale` (
  `sale_id` varchar(36) PRIMARY KEY,
  `sale_number` varchar(20) UNIQUE,
  `sale_date` datetime DEFAULT CURRENT_TIMESTAMP,
  `sale_type` ENUM('RETAIL', 'MAINTENANCE', 'INTERNAL') DEFAULT 'RETAIL',
  `status` ENUM('OPEN', 'COMPLETED', 'CANCELLED', 'REFUNDED', 'VOID') DEFAULT 'OPEN',
  `payment_status` ENUM('UNPAID', 'PARTIAL', 'PAID', 'REFUNDED') DEFAULT 'UNPAID',
  `customer_id` varchar(36) DEFAULT NULL,
  `maintenance_ticket_id` varchar(36) DEFAULT NULL,
  `subtotal_amount` decimal(10,2) DEFAULT 0.00,
  `discount_amount` decimal(10,2) DEFAULT 0.00,
  `tax_amount` decimal(10,2) DEFAULT 0.00,
  `total_amount` decimal(10,2) DEFAULT 0.00,
  `is_repair_service` boolean DEFAULT FALSE,
  `repair_service_id` varchar(36) DEFAULT NULL,
  `performed_by` varchar(36) DEFAULT NULL,
  `shipping_address_id` varchar(36) DEFAULT NULL,
  `notes` text DEFAULT NULL,
  `location_id` varchar(36) NOT NULL,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `created_by` varchar(36) DEFAULT NULL,
  `updated_by` varchar(36) DEFAULT NULL,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'PENDING',
  FOREIGN KEY (`customer_id`) REFERENCES `customer`(`customer_id`),
  FOREIGN KEY (`maintenance_ticket_id`) REFERENCES `maintenance_ticket`(`ticket_id`),
  FOREIGN KEY (`repair_service_id`) REFERENCES `repair_service`(`repair_service_id`),
  FOREIGN KEY (`performed_by`) REFERENCES `user`(`user_id`),
  FOREIGN KEY (`shipping_address_id`) REFERENCES `address`(`address_id`),
  FOREIGN KEY (`location_id`) REFERENCES `location`(`location_id`)
);

-- SALE_ITEM TABLE (Line items for invoices)
CREATE TABLE `sale_item` (
  `sale_item_id` varchar(36) PRIMARY KEY,
  `sale_id` varchar(36) NOT NULL,
  `item_id` varchar(36) DEFAULT NULL,
  `description` text DEFAULT NULL,
  `line_type` ENUM('PRODUCT', 'SPARE_PART', 'LABOR', 'FEE', 'ADJUSTMENT') DEFAULT 'PRODUCT',
  `quantity` int NOT NULL DEFAULT 1,
  `unit_price` decimal(10,2) NOT NULL,
  `discount_amount` decimal(10,2) DEFAULT 0.00,
  `tax_amount` decimal(10,2) DEFAULT 0.00,
  `line_total` decimal(10,2) NOT NULL,
  `affects_inventory` boolean DEFAULT TRUE,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'PENDING',
  FOREIGN KEY (`sale_id`) REFERENCES `sale`(`sale_id`),
  FOREIGN KEY (`item_id`) REFERENCES `item`(`item_id`)
);

-- PAYMENT TABLE (Detailed payment history per sale)
CREATE TABLE `payment` (
  `payment_id` varchar(36) PRIMARY KEY,
  `sale_id` varchar(36) NOT NULL,
  `payment_date` datetime DEFAULT CURRENT_TIMESTAMP,
  `amount` decimal(10,2) NOT NULL,
  `payment_method` ENUM('CASH', 'CARD', 'TRANSFER', 'CREDIT', 'MOBILE') DEFAULT 'CASH',
  `payment_type` ENUM('PAYMENT', 'REFUND', 'ADJUSTMENT') DEFAULT 'PAYMENT',
  `status` ENUM('PENDING', 'COMPLETED', 'FAILED', 'CANCELLED') DEFAULT 'COMPLETED',
  `transaction_reference` varchar(100) DEFAULT NULL,
  `notes` text DEFAULT NULL,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `created_by` varchar(36) DEFAULT NULL,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'PENDING',
  FOREIGN KEY (`sale_id`) REFERENCES `sale`(`sale_id`),
  FOREIGN KEY (`created_by`) REFERENCES `user`(`user_id`)
);

-- PART_REQUEST TABLE (Track who requested parts for maintenance)
CREATE TABLE `part_request` (
  `request_id` varchar(36) PRIMARY KEY,
  `ticket_id` varchar(36) NOT NULL,
  `part_id` varchar(36) NOT NULL,
  `quantity_requested` int NOT NULL DEFAULT 1,
  `requested_by` varchar(36) NOT NULL,
  `requested_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `approved_by` varchar(36) DEFAULT NULL,
  `approved_at` datetime DEFAULT NULL,
  `status` ENUM('REQUESTED', 'APPROVED', 'REJECTED', 'FULFILLED', 'CANCELLED') DEFAULT 'REQUESTED',
  `source_preference` ENUM('STORE_INVENTORY', 'EXTERNAL_SUPPLIER', 'CUSTOMER_PROVIDED') DEFAULT 'STORE_INVENTORY',
  `notes` text DEFAULT NULL,
  `location_id` varchar(36) NOT NULL,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'PENDING',
  FOREIGN KEY (`ticket_id`) REFERENCES `maintenance_ticket`(`ticket_id`),
  FOREIGN KEY (`part_id`) REFERENCES `item`(`item_id`),
  FOREIGN KEY (`requested_by`) REFERENCES `user`(`user_id`),
  FOREIGN KEY (`approved_by`) REFERENCES `user`(`user_id`),
  FOREIGN KEY (`location_id`) REFERENCES `location`(`location_id`)
);

-- PART_USAGE TABLE (Source of truth for parts consumed in maintenance)
CREATE TABLE `part_usage` (
  `usage_id` varchar(36) PRIMARY KEY,
  `ticket_id` varchar(36) NOT NULL,
  `request_id` varchar(36) DEFAULT NULL,
  `sale_item_id` varchar(36) DEFAULT NULL,
  `part_id` varchar(36) NOT NULL,
  `quantity_used` int NOT NULL DEFAULT 1,
  `unit_cost` decimal(10,2) NOT NULL,
  `total_cost` decimal(10,2) GENERATED ALWAYS AS (`quantity_used` * `unit_cost`) VIRTUAL,
  `source` ENUM('STORE_INVENTORY', 'EXTERNAL_SUPPLIER', 'CUSTOMER_PROVIDED') DEFAULT 'STORE_INVENTORY',
  `external_supplier_id` varchar(36) DEFAULT NULL,
  `external_reference` varchar(100) DEFAULT NULL,
  `used_by` varchar(36) DEFAULT NULL,
  `used_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `notes` text DEFAULT NULL,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'PENDING',
  `location_id` varchar(36) NOT NULL,
  FOREIGN KEY (`ticket_id`) REFERENCES `maintenance_ticket`(`ticket_id`),
  FOREIGN KEY (`request_id`) REFERENCES `part_request`(`request_id`),
  FOREIGN KEY (`sale_item_id`) REFERENCES `sale_item`(`sale_item_id`),
  FOREIGN KEY (`part_id`) REFERENCES `item`(`item_id`),
  FOREIGN KEY (`external_supplier_id`) REFERENCES `supplier`(`supplier_id`),
  FOREIGN KEY (`used_by`) REFERENCES `user`(`user_id`),
  FOREIGN KEY (`location_id`) REFERENCES `location`(`location_id`)
);

-- TICKET_UPDATE TABLE (Audit trail for maintenance tickets)
CREATE TABLE `ticket_update` (
  `update_id` varchar(36) PRIMARY KEY,
  `ticket_id` varchar(36) NOT NULL,
  `user_id` varchar(36) NOT NULL,
  `update_type` ENUM('STATUS_CHANGE', 'NOTE_ADDED', 'PART_USED', 'COST_UPDATED', 'TECHNICIAN_ASSIGNED') NOT NULL,
  `old_value` text DEFAULT NULL,
  `new_value` text DEFAULT NULL,
  `description` text NOT NULL,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `sync_status` ENUM('PENDING', 'SYNCED', 'CONFLICT') DEFAULT 'PENDING',
  FOREIGN KEY (`ticket_id`) REFERENCES `maintenance_ticket`(`ticket_id`),
  FOREIGN KEY (`user_id`) REFERENCES `user`(`user_id`)
);

-- ========== SYNC & OFFLINE SUPPORT TABLES ==========

-- SYNC_QUEUE TABLE
CREATE TABLE `sync_queue` (
  `queue_id` varchar(36) PRIMARY KEY,
  `table_name` varchar(50) NOT NULL,
  `record_id` varchar(36) NOT NULL,
  `operation` ENUM('INSERT', 'UPDATE', 'DELETE') NOT NULL,
  `data` json NOT NULL,
  `sync_status` ENUM('PENDING', 'PROCESSING', 'SYNCED', 'FAILED', 'CONFLICT') DEFAULT 'PENDING',
  `retry_count` int DEFAULT 0,
  `error_message` text DEFAULT NULL,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `processed_at` datetime DEFAULT NULL,
  `location_id` varchar(36) NOT NULL,
  `priority` ENUM('LOW', 'MEDIUM', 'HIGH') DEFAULT 'MEDIUM',
  `conflict_resolution` ENUM('AUTO', 'MANUAL') DEFAULT 'AUTO',
  INDEX `idx_sync_status` (`sync_status`),
  INDEX `idx_location_id` (`location_id`),
  FOREIGN KEY (`location_id`) REFERENCES `location`(`location_id`)
);

-- SYNC_CONFLICT TABLE
CREATE TABLE `sync_conflict` (
  `conflict_id` varchar(36) PRIMARY KEY,
  `queue_id` varchar(36) NOT NULL,
  `table_name` varchar(50) NOT NULL,
  `record_id` varchar(36) NOT NULL,
  `local_data` json NOT NULL,
  `remote_data` json NOT NULL,
  `conflict_type` ENUM('DATA_CONFLICT', 'VERSION_CONFLICT', 'DELETE_CONFLICT') NOT NULL,
  `resolution` ENUM('LOCAL_WINS', 'REMOTE_WINS', 'MANUAL_MERGE', 'NEW_RECORD') DEFAULT NULL,
  `resolved_data` json DEFAULT NULL,
  `resolved_by` varchar(36) DEFAULT NULL,
  `resolved_at` datetime DEFAULT NULL,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `sync_status` ENUM('PENDING', 'RESOLVED', 'IGNORED') DEFAULT 'PENDING',
  FOREIGN KEY (`queue_id`) REFERENCES `sync_queue`(`queue_id`),
  FOREIGN KEY (`resolved_by`) REFERENCES `user`(`user_id`)
);

-- SYNC_HISTORY TABLE (Audit trail)
CREATE TABLE `sync_history` (
  `history_id` varchar(36) PRIMARY KEY,
  `sync_session_id` varchar(36) NOT NULL,
  `location_id` varchar(36) NOT NULL,
  `start_time` datetime NOT NULL,
  `end_time` datetime DEFAULT NULL,
  `records_sent` int DEFAULT 0,
  `records_received` int DEFAULT 0,
  `conflicts_found` int DEFAULT 0,
  `status` ENUM('SUCCESS', 'PARTIAL', 'FAILED') DEFAULT 'SUCCESS',
  `error_message` text DEFAULT NULL,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_sync_session` (`sync_session_id`),
  INDEX `idx_sync_time` (`start_time`),
  FOREIGN KEY (`location_id`) REFERENCES `location`(`location_id`)
);

-- ORDER_ITEM TABLE
CREATE TABLE `order_item` (
  `order_item_id` varchar(36) PRIMARY KEY,
  `order_id` varchar(36) NOT NULL,
  `item_id` varchar(36) NOT NULL,
  `quantity` int NOT NULL,
  `unit_price` decimal(10,2) NOT NULL,
  `total_price` decimal(10,2) GENERATED ALWAYS AS (`quantity` * `unit_price`) VIRTUAL,
  `received_quantity` int DEFAULT 0,

-- ========== STORED PROCEDURES ==========

-- Procedure to update inventory after sale item
DELIMITER $$
CREATE PROCEDURE `UpdateInventoryAfterSaleItem`(
    IN p_item_id VARCHAR(36),
    IN p_location_id VARCHAR(36),
    IN p_quantity INT,
    IN p_sale_item_id VARCHAR(36),
    IN p_user_id VARCHAR(36)
)
BEGIN
    DECLARE current_qty INT;
    DECLARE new_qty INT;
    
    -- Get current quantity
    SELECT quantity INTO current_qty 
    FROM inventory 
    WHERE item_id = p_item_id AND location_id = p_location_id;
    
    -- Calculate new quantity
    SET new_qty = current_qty - p_quantity;
    
    -- Update inventory
    UPDATE inventory 
    SET quantity = new_qty, 
        updated_at = NOW(),
        updated_by = p_user_id
    WHERE item_id = p_item_id AND location_id = p_location_id;
    
    -- Record stock movement
    INSERT INTO stock_movement (
        movement_id, item_id, location_id, movement_type, 
        quantity, previous_quantity, new_quantity,
        reference_id, reference_type, performed_by
    ) VALUES (
        UUID(), p_item_id, p_location_id, 'SALE',
        p_quantity, current_qty, new_qty,
        p_sale_item_id, 'SALE_ITEM', p_user_id
    );
END$$
DELIMITER ;

-- Procedure to check low stock items
DELIMITER $$
CREATE PROCEDURE `CheckLowStockItems`(IN p_location_id VARCHAR(36))
BEGIN
    SELECT 
        i.item_id,
        i.name,
        i.sku,
        i.item_type,
        i.min_stock_level,
        inv.quantity as current_stock,
        CASE 
            WHEN inv.quantity <= i.min_stock_level THEN 'CRITICAL'
            WHEN inv.quantity <= (i.min_stock_level * 2) THEN 'LOW'
            ELSE 'OK'
        END as stock_status
    FROM item i
    JOIN inventory inv ON i.item_id = inv.item_id
    WHERE inv.location_id = p_location_id
      AND inv.quantity <= (i.min_stock_level * 2)
    ORDER BY stock_status, inv.quantity ASC;
END$$
DELIMITER ;

-- ========== VIEWS ==========

-- View for inventory summary
CREATE VIEW `inventory_summary` AS
SELECT 
    i.item_id,
    i.name,
    i.sku,
    i.item_type,
    c.name as category,
    i.min_stock_level,
    i.reorder_quantity,
    SUM(inv.quantity) as total_quantity,
    GROUP_CONCAT(CONCAT(l.name, ': ', inv.quantity)) as location_breakdown
FROM item i
LEFT JOIN inventory inv ON i.item_id = inv.item_id
LEFT JOIN location l ON inv.location_id = l.location_id
LEFT JOIN category c ON i.category_id = c.category_id
GROUP BY i.item_id, i.name, i.sku, i.item_type, c.name, i.min_stock_level, i.reorder_quantity;

-- View for parts usage report (retail vs maintenance)
CREATE VIEW `parts_usage_report` AS
SELECT 
    i.item_id,
    i.name as part_name,
    i.sku,
    COALESCE(retail.retail_quantity, 0) as retail_quantity,
    COALESCE(maintenance.maintenance_quantity, 0) as maintenance_quantity,
    COALESCE(retail.retail_revenue, 0) as retail_revenue,
    COALESCE(maintenance.technicians_used_by, 0) as technicians_used_by,
    maintenance.technician_names as technician_names
FROM item i
LEFT JOIN (
    SELECT 
        si.item_id,
        SUM(si.quantity) as retail_quantity,
        SUM(si.line_total) as retail_revenue
    FROM sale_item si
    JOIN sale s ON s.sale_id = si.sale_id
    WHERE s.sale_type = 'RETAIL'
    GROUP BY si.item_id
) retail ON retail.item_id = i.item_id
LEFT JOIN (
    SELECT 
        pu.part_id as item_id,
        SUM(pu.quantity_used) as maintenance_quantity,
        COUNT(DISTINCT pu.used_by) as technicians_used_by,
        GROUP_CONCAT(DISTINCT u.first_name) as technician_names
    FROM part_usage pu
    LEFT JOIN user u ON pu.used_by = u.user_id
    GROUP BY pu.part_id
) maintenance ON maintenance.item_id = i.item_id
WHERE i.item_type = 'SPARE_PART';

-- View for maintenance tickets with details
CREATE VIEW `maintenance_tickets_view` AS
SELECT 
    mt.ticket_id,
    mt.ticket_number,
    mt.status,
    mt.priority,
    c.name as customer_name,
    c.phone as customer_phone,
    COALESCE(cd.model, i.name) as device_model,
    cd.serial_number as device_serial,
    CONCAT(u.first_name, ' ', u.last_name) as technician_name,
    mt.problem_description,
    mt.total_cost,
    mt.created_at,
    mt.estimated_completion,
    mt.location_id
FROM maintenance_ticket mt
LEFT JOIN customer c ON mt.customer_id = c.customer_id
LEFT JOIN customer_device cd ON mt.customer_device_id = cd.device_id
LEFT JOIN item i ON cd.catalog_item_id = i.item_id
LEFT JOIN user u ON mt.technician_id = u.user_id;

-- ========== TRIGGERS ==========

-- Trigger to generate sale number
DELIMITER $$
CREATE TRIGGER `before_sale_insert`
BEFORE INSERT ON `sale`
FOR EACH ROW
BEGIN
    IF NEW.sale_number IS NULL THEN
        SET NEW.sale_number = CONCAT('SALE-', DATE_FORMAT(NOW(), '%Y%m%d-'), LPAD(FLOOR(RAND() * 10000), 4, '0'));
    END IF;
END$$
DELIMITER ;

-- Trigger to generate ticket number
DELIMITER $$
CREATE TRIGGER `before_ticket_insert`
BEFORE INSERT ON `maintenance_ticket`
FOR EACH ROW
BEGIN
    IF NEW.ticket_number IS NULL THEN
        SET NEW.ticket_number = CONCAT('TICKET-', DATE_FORMAT(NOW(), '%Y%m%d-'), LPAD(FLOOR(RAND() * 10000), 4, '0'));
    END IF;
END$$
DELIMITER ;

-- Trigger to auto-add to sync queue on insert
DELIMITER $$
CREATE TRIGGER `after_sale_insert_sync`
AFTER INSERT ON `sale`
FOR EACH ROW
BEGIN
    INSERT INTO sync_queue (
        queue_id, table_name, record_id, operation, 
        data, sync_status, location_id, created_at
    ) VALUES (
        UUID(), 'sale', NEW.sale_id, 'INSERT',
        JSON_OBJECT(
            'sale_id', NEW.sale_id,
            'sale_number', NEW.sale_number,
            'sale_date', NEW.sale_date,
            'sale_type', NEW.sale_type,
            'status', NEW.status,
            'payment_status', NEW.payment_status,
            'total_amount', NEW.total_amount,
            'location_id', NEW.location_id,
            'customer_id', NEW.customer_id,
            'maintenance_ticket_id', NEW.maintenance_ticket_id
        ),
        'PENDING', NEW.location_id, NOW()
    );
END$$
DELIMITER ;
